Array.prototype.some()

Find out if at least one person 19 or older?

const isAdult = people.some(function (person) {
const currentYear = new Date().getFullYear();
if (currentYear - person.year >= 19) {
return true;
}
});
console.log(isAdult);

- returns true = as there is someone older than 19

console.log({ isAdult });

returns the object

Simpler version using fat arrow

const isAdultSimple = people.some((person) => {
const currentYear = new Date().getFullYear();
return currentYear - person.year >= 19;
});
console.log(isAdultSimple);

Fat Arrow with implicit return - Breakdown even further

const isAdultShort = people.some(
(person) => new Date().getFullYear() - person.year >= 19 );
console.log(isAdultShort);

Array.prototype.every()

Find out if everyone 19 or older

const allAdults = people.every(
(person) => new Date().getFullYear() - person.year >= 19 );
console.log(allAdults);

Array.prototype.find()

Find is like filter, but instead returns just the one you are looking for
Find the comment with the ID of 823423

const comment = comments.find(function (comment) {
if (comment.id === 823423) {
return true; } });
console.log(comment);

Change above to arrow function with implicit return

const arrowComment = comments.find((comment) => comment.id === 823423);
console.log(arrowComment);

Dont need the if as will return boolean

Array.prototype.findIndex()

Find the comment with this ID

const index = comments.findIndex((comment) => comment.id === 823423);
console.log(index);

Array.prototype.splice()

Delete the comment with the ID of 823423

comments.splice(index, 1);
console.table(comments);

Array.prototype.slice()

Create a new array without the deleted item

previous array remains in tact

const newComments = [

];
